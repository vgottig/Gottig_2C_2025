85void distanciasTask(void *pvParameter)
86{
87 uint8_t distancia;
88
89 while (true)
90 {
91 if (tecla1) // Si la medición está activada, lee la distancia
92 {
93 distancia = HcSr04ReadDistanceInCentimeters(); // Para que me de la distancia en cm
94
95 if (!tecla2)
96 { // Si la tecla 2 no está pulsada
97 LcdItsE0803Write(distancia); // Muestra la distancia en el LCD
98 }
99
100 if (distancia < 10) // Si la distancia es menor a 10 cm se apagan todos los leds
101 {
102 LedsOffAll();
103 }
104
105 else if (distancia > 10 && distancia < 20) // Si la distancia está entre
106 { // 10 y 20, se prende el led 1 y
107 LedOn(LED_1); // se apagan los led 2 y led 3
108 LedOff(LED_2);
109 LedOff(LED_3);
110 }
111
112 else if (distancia > 20 && distancia < 30) // Si la distancia está entre 20 y 30
113 { // se prenden los led 1 y 2
114 LedOn(LED_1); // y se apaga el led 3
115 LedOn(LED_2);
116 LedOff(LED_3);
117 }
118
119 else if (distancia > 30) // Si la distancia es mayor a 30
120 { // se prenden los tres leds
121 LedOn(LED_1);
122 LedOn(LED_2);
123 LedOn(LED_3);
124 }
125 }
126 else
127 {
128 LcdItsE0803Off(); // Apaga el LCD
129 LedsOffAll(); // Apaga los tres leds
130 }
131
132 vTaskDelay(CONFIG_BLINK_PERIOD / portTICK_PERIOD_MS); // Cada 1000 ms
133 }
134}

141void controlTeclasTask(void *pvParameter)
142{
143 uint8_t teclas;
144
145 while (true)
146 {
147 teclas = SwitchesRead(); // Le asigno el estado de las teclas (switches)
148
149 switch (teclas) // Reviso el estado de las teclas
150 {
151 case SWITCH_1:
152 tecla1 = !tecla1; // Si se pulsa la tecla 1, ahora vale true. Mide y muestra
153 break;
154 case SWITCH_2:
155 tecla2 = !tecla2; // Si se pulsa la tecla 2, ahora vale true
156 // LCD mantiene fijo el valor aunque el sensor siga midiendo
157 break;
158 }
159
160 vTaskDelay(CONFIG_BLINK_PERIOD_2 / portTICK_PERIOD_MS); // Cada 50 ms
161 }
162}
