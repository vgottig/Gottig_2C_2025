86static void medicion_vehiculos(void *pvParameters)
87{
88 while (1)
89 {
90 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // Espera la notificación del timer
91
92 distancia_vehiculos = HcSr04ReadDistanceInCentimeters();
93
94 distancia_vehiculos = distancia_vehiculos * 100; // Convierto los valores a metros
95
96 if (distancia_vehiculos > 5)
97 LedOn(LED_1); // LED verde
98
99 else
100
101 LedsOffAll();
102
103 if (distancia_vehiculos < 5 && distancia_vehiculos > 3)
104 {
105 LedOn(LED_1); // LED verde y
106 LedOn(LED_2); // LED amarillo
107 UartSendString(UART_CONNECTOR, "Precaución, vehículo cerca "); // UART_CONNECTOR es un UART externo por los pines GPIO18 y GPIO19
108 UartSendString(UART_CONNECTOR, "\r\n");
109
110 while (distancia_vehiculos < 5 && distancia_vehiculos > 3)
111 {
112 GPIOOn(Buzzer);
113 vTaskDelay(pdMS_TO_TICKS(1000)); // Espera 1 segundo
114 GPIOOff(Buzzer);
115 }
116 }
117 else
118 LedsOffAll();
119
120 if (distancia_vehiculos < 3)
121 {
122 LedOn(LED_1); // LED verde
123 LedOn(LED_2); // LED amarillo y
124 LedOn(LED_3); // LED rojo
125 UartSendString(UART_CONNECTOR, "Peligro, vehículo cerca");
126 UartSendString(UART_CONNECTOR, "\r\n");
127
128 while (distancia_vehiculos < 3)
129 {
130 GPIOOn(Buzzer);
131 vTaskDelay(pdMS_TO_TICKS(500)); // Espera 0.5 segundos
132 GPIOOff(Buzzer);
133 }
134 }
135 else
136 LedsOffAll();
137 }

145static void detector_caidas(void *pvParameters)
146{
147 while (1)
148 {
149 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
150
151 AnalogInputReadSingle(CH0, &lectura_eje_X);
152 AnalogInputReadSingle(CH1, &lectura_eje_y);
153 AnalogInputReadSingle(CH2, &lectura_eje_z);
154
155 sumatoria_ejes = lectura_eje_X + lectura_eje_y + lectura_eje_z;
156
157 if(sumatoria_ejes > umbral_caida)
158 {
159 UartSendString(UART_CONNECTOR, "Caída detectada");
160 UartSendString(UART_CONNECTOR, "\r\n");
161 }
162
163 }
164}
