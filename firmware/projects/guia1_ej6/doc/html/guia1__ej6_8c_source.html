71int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t * bcd_number) {
72 // Inicializar el array de salida con ceros
73 // Esto es útil si el número de entrada tiene menos dígitos que 'digits'.
74 for (int i = 0; i < digits; ++i) {
75 bcd_number[i] = 0;
76 }
77
78 // Usar un índice para rellenar el array de derecha a izquierda
79 int8_t index = digits - 1;
80
81 // Bucle para extraer cada dígito del número
82 while (data > 0 && index >= 0) {
83 // Obtener el último dígito (el residuo de la división por 10)
84 uint8_t digit = data % 10;
85
86 // Guardar el dígito en el array
87 bcd_number[index] = digit;
88
89 // Moverse al siguiente dígito
90 data /= 10;
91 index--;
92 }
93
94 return 0; // Devuelve 0 para indicar que la operación fue exitosa
95}

103void functionBCD(uint8_t digit, gpioConf_t *vector){ // Recibe un dígito (valor numérico)
104 // y un vector que contiene la configuración de b0 a b3
105 for(uint8_t i=0; i<4; i++){
106 GPIOInit(vector[i].pin,vector[i].dir); // Inicializa cada pin en la dirección indicada del vector
107 }
108
109 // Evalúa cada bit
110 for(int i=0; i<4; i++){
111 if((digit&(1<<i))==0){ // Comprueba si el bit i de digit está en 0 o en 1. << (shifteo/corrimiento a la derecha)
112 GPIOOff(vector[i].pin); // Si está en 0 llama a GPIOOff y escribe nivel bajo
113 }
114 else
115 GPIOOn(vector[i].pin); // Si está en 1 llama a GPIOOn y escribe nivel alto
116 }
117}
